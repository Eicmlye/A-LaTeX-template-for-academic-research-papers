\makeatletter
	\def\clearintentionalblank{
		\clearpage
		\if@twoside
			\ifodd
				\c@page
			\else
				\hbox{}
				\vfill
				\begin{center}
					This page intentionally left blank.
				\end{center}
				\vfill
				\thispagestyle{empty}
				\newpage
				\if@twocolumn
					\hbox{}
					\newpage
				\fi
			\fi
		\fi
		}
		
	\def\clearblankpage{
		\clearpage
		\if@twoside
			\ifodd
				\c@page
			\else
				\hbox{}
				\vfill
				\begin{center}
					~
				\end{center}
				\vfill
				\thispagestyle{empty}
				\newpage
				\if@twocolumn
					\hbox{}
					\newpage
				\fi
			\fi
		\fi
		}
\makeatother
